9    pub timings: GenTimings,
10    pub state_root: Cell,

13impl ParsedProofs {
14    pub(crate) fn get_last_transaction_id(
15        &self,
16        account: &HashBytes,
17    ) -> anyhow::Result<LastTransactionId> {
18        type ShardAccountsShort = Dict<HashBytes, TonShardAccountShort>;
19
20        let proof = self
21            .state_root
22            .parse::<TonShardStateShort>()
23            .context("invalid state proof")?;
24
25        let accounts = proof
26            .accounts
27            .parse::<ShardAccounts>()
28            .context("failed to parse shard accounts")?;
29        let accounts = ShardAccountsShort::from_raw(accounts.dict().root().clone());
30
31        let Some(state) = accounts.get(account).context("failed to get tx id")? else {
32            anyhow::bail!("account state not found");
33        };
34
35        Ok(LastTransactionId {
36            hash: state.last_trans_hash,
37            lt: state.last_trans_lt,
38        })
39    }
40}

61impl<'a> Load<'a> for TonShardAccountShort {
62    fn load_from(slice: &mut CellSlice<'a>) -> anyhow::Result<Self, Error> {
63        // Skip `split_depth`
64        slice.skip_first(5, 0)?;
65        // Skip balance.
66        _ = CurrencyCollection::load_from(slice)?;
67        // Skip account.
68        Cell::load_from(slice)?;
69
70        Ok(Self {
71            last_trans_hash: slice.load_u256()?,
72            last_trans_lt: slice.load_u64()?,
73        })
74    }
