6#[derive(Deserialize)]
7#[serde(rename_all = "camelCase", tag = "type")]
8pub enum ContractState {
9    NotExists {
10        timings: GenTimings,
11    },
12    #[serde(rename_all = "camelCase")]
13    Exists {
14        #[serde(deserialize_with = "deserialize_account")]
15        account: Box<Account>,
16        timings: GenTimings,
17        last_transaction_id: LastTransactionId,
18    },
19    Unchanged {
20        timings: GenTimings,
21    },
22}

24fn deserialize_account<'de, D>(deserializer: D) -> Result<Box<Account>, D::Error>
25where
26    D: serde::Deserializer<'de>,
27{
28    use serde::de::Error;
29    use tycho_types::cell::Load;
30
31    fn read_account(cell: Cell) -> Result<Box<Account>, tycho_types::error::Error> {
32        let s = &mut cell.as_slice()?;
33        Ok(Box::new(Account {
34            address: <_>::load_from(s)?,
35            storage_stat: <_>::load_from(s)?,
36            last_trans_lt: <_>::load_from(s)?,
37            balance: <_>::load_from(s)?,
38            state: <_>::load_from(s)?,
39        }))
40    }
41
42    Boc::deserialize(deserializer).and_then(|cell| read_account(cell).map_err(Error::custom))
43}
