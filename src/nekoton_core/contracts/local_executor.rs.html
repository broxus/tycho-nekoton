1use anyhow::Result;
2use everscale_types::cell::{Cell, CellBuilder, HashBytes};
3use everscale_types::models::{
4    Account, BlockchainConfig, ComputePhase, IntAddr, LibDescr, OutAction, OutActionsRevIter,
5    OwnedMessage, OwnedRelaxedMessage,
6};
7use everscale_types::prelude::{CellFamily, Dict, Store};
8
9use everscale_types::num::Tokens;
10use tycho_executor::phase::{ComputePhaseContext, TransactionInput};
11use tycho_executor::{ExecutorParams, ParsedConfig};
12use tycho_vm::OwnedCellSlice;
13
14pub struct ComputePhaseResult {
15    pub exit_code: i32,
16    pub success: bool,
17    pub out_messages: Vec<OwnedRelaxedMessage>,
18}
19#[allow(clippy::too_many_arguments)]
20pub fn execute_message(
21    block_unixtime: u32,
22    block_lt: u64,
23    account: Account,
24    message: OwnedMessage,
25    config: BlockchainConfig,
26    rand_seed: HashBytes,
27    libraries: Dict<HashBytes, LibDescr>,
28    vm_modifiers: tycho_vm::BehaviourModifiers,
29) -> Result<ComputePhaseResult> {
30    let mut builder = CellBuilder::new();
31    message.store_into(&mut builder, Cell::empty_context())?;
32    let in_msg_cell = builder.build()?;
33
34    let executor_params = ExecutorParams {
35        libraries,
36        rand_seed,
37        block_unixtime,
38        block_lt,
39        vm_modifiers,
40        ..Default::default()
41    };
42
43    let config = ParsedConfig::parse(config, block_unixtime)?;
44
45    let executor = tycho_executor::Executor::new(&executor_params, &config);
46
47    let IntAddr::Std(std_addr) = &account.address.clone() else {
48        anyhow::bail!("Invalid address type");
49    };
50
51    let mut state = executor.begin(std_addr, Some(account))?;
52    let received_message = state.receive_in_msg(in_msg_cell)?;
53
54    let compute_phase_result = state.compute_phase(ComputePhaseContext {
55        input: TransactionInput::Ordinary(&received_message),
56        storage_fee: Tokens::ZERO,
57        force_accept: true,
58        inspector: None,
59    })?;
60
61    let executed_compute_phase = match compute_phase_result.compute_phase {
62        ComputePhase::Skipped(result) => {
63            anyhow::bail!("Compute phase is skipped.Reason: {:?}", result.reason)
64        }
65        ComputePhase::Executed(executed_result) => executed_result,
66    };
67
68    let mut msgs = Vec::new();
69    let actions_slice = OwnedCellSlice::new_allow_exotic(compute_phase_result.actions);
70    let out_actions_iter = OutActionsRevIter::new(actions_slice.apply());
71    for action in out_actions_iter.flatten() {
72        if let OutAction::SendMsg { out_msg, .. } = action {
73            msgs.push(out_msg.load()?);
74        }
75    }
76
77    msgs.reverse();
78
79    Ok(ComputePhaseResult {
80        exit_code: !executed_compute_phase.exit_code,
81        success: executed_compute_phase.success,
82        out_messages: msgs,
83    })
84}
