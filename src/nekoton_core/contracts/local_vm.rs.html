12pub struct LocalVmBuilder {
13    libraries: Dict<HashBytes, LibDescr>,
14    behaviour_modifiers: Option<BehaviourModifiers>,
15    unpacked_config: Option<UnpackedConfig>,
16}
17
18impl Default for LocalVmBuilder {
19    fn default() -> Self {
20        Self::new()
21    }
22}
23
24impl LocalVmBuilder {
25    pub fn new() -> Self {
26        Self {
27            libraries: Dict::default(),
28            behaviour_modifiers: None,
29            unpacked_config: None,
30        }
31    }
32
33    pub fn with_libraries(mut self, libraries: Dict<HashBytes, LibDescr>) -> Self {
34        self.libraries = libraries;
35        self
36    }
37    pub fn with_behaviour_modifiers(mut self, modifiers: BehaviourModifiers) -> Self {
38        self.behaviour_modifiers = Some(modifiers);
39        self
40    }
41
42    pub fn with_unpacked_config(mut self, config: BlockchainConfig, now: u32) -> Result<Self> {
43        let partial_config = SmcInfoTonV6::unpack_config_partial(&config, now)?;
44        self.unpacked_config = Some(partial_config);
45        Ok(self)
46    }
47
48    pub fn build(self) -> Result<LocalVm> {
49        Ok(LocalVm {
50            libraries: self.libraries,
51            behaviour_modifiers: self.behaviour_modifiers.unwrap_or_default(),
52            config: match self.unpacked_config {
53                Some(config) => config,
54                None => anyhow::bail!("failed to build unpacked config"),
55            },
56        })
57    }
58}

60pub struct LocalVm {
61    libraries: Dict<HashBytes, LibDescr>,
62    behaviour_modifiers: BehaviourModifiers,
63    config: UnpackedConfig,
64}
65
66impl LocalVm {
67    pub fn call_getter(
68        &self,
69        block_unixtime: u32,
70        block_lt: u64,
71        account: &Account,
72        stack: Vec<RcStackValue>,
73    ) -> Result<VmGetterOutput, ExecutionError> {
74        let state = match &account.state {
75            AccountState::Active(state_init) => state_init,
76            _ => return Err(ExecutionError::AccountNotActive(account.address.clone())),
77        };
78        let code = state
79            .clone()
80            .code
81            .ok_or(ExecutionError::AccountHasNoCode(account.address.clone()))?;
82
83        let smc = SmcInfoBase::new()
84            .with_account_addr(account.address.clone())
85            .with_account_balance(account.balance.clone())
86            .with_block_lt(block_lt)
87            .with_tx_lt(block_lt)
88            .with_now(block_unixtime)
89            .require_ton_v4()
90            .require_ton_v6()
91            .with_unpacked_config(self.config.clone().into_tuple())
92            .require_ton_v11();
93
94        let data = state.clone().data.unwrap_or_default();
95
96        let mut vm_state = VmStateBuilder::new()
97            .with_code(code)
98            .with_data(data)
99            .with_stack(stack)
100            .with_smc_info(smc)
101            .with_libraries(&self.libraries)
102            .with_modifiers(self.behaviour_modifiers)
103            .with_gas(GasParams::getter())
104            .build();
105
106        let exit_code = !vm_state.run();
107
108        Ok(VmGetterOutput {
109            exit_code,
110            stack: vm_state.stack.items.clone(),
111            success: exit_code == 0 || exit_code == 1,
112        })
113    }
114}
