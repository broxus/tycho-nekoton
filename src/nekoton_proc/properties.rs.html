1use case::CaseExt;
2
3#[derive(Clone)]
4pub enum StructProperty {
5    Simple {
6        name: Option<String>,
7        type_name: syn::Type,
8    },
9    Tuple {
10        name: String,
11        //_fields: Vec<StructProperty>,
12    },
13    Array {
14        name: String,
15        internal: Box<StructProperty>,
16    },
17    Option {
18        name: String,
19        internal: Box<StructProperty>,
20    },
21    HashMap {
22        name: String,
23        key: Box<StructProperty>,
24        value: Box<StructProperty>,
25    },
26}
27
28impl StructProperty {
29    pub fn type_name_quote(&self) -> syn::Type {
30        match self {
31            StructProperty::Simple { type_name, .. } => type_name.clone(),
32            StructProperty::Tuple { name, .. } => syn::parse_str(&name.to_camel()).unwrap(),
33            StructProperty::Array { internal, .. } => {
34                let ty = internal.type_name_quote();
35                syn::parse_quote!(Vec<#ty>)
36            }
37            StructProperty::Option { internal, .. } => {
38                let ty = internal.type_name_quote();
39                syn::parse_quote!(Option<#ty>)
40            }
41            StructProperty::HashMap { key, value, .. } => {
42                let key = key.type_name_quote();
43                let value = value.type_name_quote();
44                syn::parse_quote!(std::collections::HashMap<#key, #value>)
45            }
46        }
47    }
48
49    pub fn name(&self) -> String {
50        match self {
51            StructProperty::Simple { name, .. } => {
52                let name = name.clone();
53                name.unwrap_or_default()
54            }
55            StructProperty::Tuple { name, .. } => name.clone(),
56            StructProperty::Array { name, .. } => name.clone(),
57            StructProperty::Option { name, .. } => name.clone(),
58            StructProperty::HashMap { name, .. } => name.clone(),
59        }
60    }
61}
