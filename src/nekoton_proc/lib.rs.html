31#[proc_macro_attribute]
32pub fn abi(params: TokenStream, input: TokenStream) -> TokenStream {
33    let mut generated_structs: Vec<proc_macro2::TokenStream> = Vec::new();
34    let mut generated_functions: Vec<proc_macro2::TokenStream> = Vec::new();
35    let mut generated_events: Vec<proc_macro2::TokenStream> = Vec::new();
36
37    let params = parse_macro_input!(params as ModuleParams);
38    let path_ident = params.path.trim_matches('"');
39    let path = std::env::current_dir()
40        .map_err(|e| format!("Failed to get current directory: {}", e))
41        .unwrap()
42        .join(path_ident);
43
44    let content = fs::read_to_string(path).unwrap();
45    let contract = serde_json::from_str::<Contract>(&content).unwrap();
46
47    let input = parse_macro_input!(input as ItemMod);
48    let mod_name = &input.ident;
49
50    let mut struct_gen = StructGen::new();
51
52    contract.functions.iter().for_each(|(_, function)| {
53        let FunctionDescriptionTokens {
54            body,
55            input,
56            output,
57            inner_models,
58        } = struct_gen.process_function(function);
59
60        generated_functions.push(body);
61
62        generated_structs.push(input);
63        generated_structs.push(output);
64        generated_structs.extend_from_slice(inner_models.as_slice());
65    });
66
67    contract.events.iter().for_each(|(_, event)| {
68        let FunctionDescriptionTokens {
69            body,
70            input,
71            output,
72            inner_models,
73        } = struct_gen.process_event(event);
74
75        generated_events.push(body);
76
77        generated_structs.push(input);
78        generated_structs.push(output);
79        generated_structs.extend_from_slice(inner_models.as_slice());
80    });
81
82    let trait_gen = train_gen::TraitImplGen::new();
83
84    let mut trait_implementations: Vec<proc_macro2::TokenStream> = Vec::new();
85
86    for (name, properties) in struct_gen.generated_structs.iter() {
87        let struct_traits = trait_gen.implement_traits(name, properties.as_slice());
88        trait_implementations.push(struct_traits)
89    }
90
91    let header_type: syn::Type =
92        syn::parse_str("everscale_types::contracts::AbiHeaderType").unwrap();
93    let abi_type: syn::Type = syn::parse_str("everscale_types::contracts::AbiVersion").unwrap();
94
95    let mut header_idents = Vec::<proc_macro2::TokenStream>::new();
96    for i in contract.headers.iter() {
97        let ty = match i {
98            AbiHeaderType::Expire => "everscale_types::contracts::AbiHeaderType::Expire",
99            AbiHeaderType::PublicKey => "everscale_types::contracts::AbiHeaderType::PublicKey",
100            AbiHeaderType::Time => "everscale_types::contracts::AbiHeaderType::Time",
101        };
102        let ty: syn::Type = syn::parse_str(ty).expect("Failed to parse header type");
103        let quote = quote! {
104            #ty
105        };
106        header_idents.push(quote);
107    }
108
109    let slice_token = quote! {
110        [ #(#header_idents),* ]
111    };
112
113    let header_count = contract.headers.len();
114    let major = contract.abi_version.major;
115    let minor = contract.abi_version.minor;
116
117    let quote = quote! {
118
119        pub mod #mod_name {
120            use anyhow::Result;
121            use everscale_types::abi::{NamedAbiType, AbiType, WithAbiType, IntoAbi, IntoPlainAbi,
122                FromAbiIter, FromAbi, AbiValue, NamedAbiValue, Function, Event
123            };
124            use num_bigint::{BigInt, BigUint};
125
126            #(#generated_structs)*
127
128            #(#trait_implementations)*
129
130            pub mod functions {
131                use super::*;
132
133                const HEADERS: [#header_type; #header_count] = #slice_token;
134                const ABI_VERSION: #abi_type = <#abi_type>::new(#major, #minor);
135
136                #(#generated_functions)*
137            }
138
139             pub mod events {
140                use super::*;
141
142                const ABI_VERSION: #abi_type = <#abi_type>::new(#major, #minor);
143
144                #(#generated_events)*
145            }
146        }
147    };
148
149    quote.into()
150}

163        Self {

171    fn process_function(&mut self, function: &Function) -> FunctionDescriptionTokens {
172        let input_token =
173            self.make_function_input_struct(function.name.as_ref(), function.inputs.clone(), false);
174        let output_token =
175            self.make_function_output_struct(function.name.as_ref(), function.outputs.clone());
176
177        let mut inner_modes = Vec::new();
178
179        let func = self.generate_func_body(function.name.as_ref(), false);
180
181        for i in self.temporary_internal_structs_idents.iter() {
182            inner_modes.push(i.clone());
183        }
184
185        self.temporary_internal_structs_idents.clear();
186
187        FunctionDescriptionTokens {
188            body: func,
189            input: input_token,
190            output: output_token,
191            inner_models: inner_modes,
192        }
193    }
194
195    fn process_event(&mut self, event: &Event) -> FunctionDescriptionTokens {
196        let input_token =
197            self.make_function_input_struct(event.name.as_ref(), event.inputs.clone(), true);
198
199        let mut inner_modes = Vec::new();
200
201        let func = self.generate_func_body(event.name.as_ref(), true);
202        for i in self.temporary_internal_structs_idents.iter() {
203            inner_modes.push(i.clone());
204        }
205
206        self.temporary_internal_structs_idents.clear();
207
208        FunctionDescriptionTokens {
209            body: func,
210            input: input_token,
211            output: proc_macro2::TokenStream::default(), //event has no output
212            inner_models: inner_modes,
213        }
214    }

216    fn generate_func_body(&self, name: &str, is_event: bool) -> proc_macro2::TokenStream {
217        let snake_function_name = name.to_snake();
218        let camel_function_name = name.to_camel();
219
220        let function_name_ident = if is_event {
221            format_ident!("{}_event", snake_function_name)
222        } else {
223            format_ident!("{}", snake_function_name)
224        };
225
226        let input_name = if is_event {
227            format!("{}EventInput", &camel_function_name)
228        } else {
229            format!("{}FunctionInput", &camel_function_name)
230        };
231
232        let inputs: Vec<_> = self
233            .generated_structs
234            .get(&input_name)
235            .cloned()
236            .unwrap_or_default()
237            .into_iter()
238            .map(|x| {
239                let name = x.name.as_ref();
240                let quote_abi_type = quote_abi_type(&x.ty);
241                quote! {
242                    NamedAbiType::new(#name, #quote_abi_type)
243                }
244            })
245            .collect();
246
247        let inputs_count = inputs.len();
248        let inputs_array = quote! {
249            [ #(#inputs),* ]
250        };
251
252        if is_event {
253            quote! {
254                pub fn #function_name_ident() -> &'static everscale_types::abi::Event {
255                    static ONCE: std::sync::OnceLock<everscale_types::abi::Event> = std::sync::OnceLock::new();
256                    ONCE.get_or_init(|| {
257                        let inputs: [NamedAbiType; #inputs_count] = #inputs_array;
258                        everscale_types::abi::Event::builder(ABI_VERSION, #name)
259                            .with_inputs(inputs)
260                            .build()
261                    })
262                }
263            }
264        } else {
265            let outputs: Vec<_> = self
266                .generated_structs
267                .get(&format!("{}FunctionOutput", &camel_function_name))
268                .cloned()
269                .unwrap_or_default()
270                .into_iter()
271                .map(|x| {
272                    let name = x.name.as_ref();
273                    let quote_abi_type = quote_abi_type(&x.ty);
274                    quote! {
275                        NamedAbiType::new(#name, #quote_abi_type)
276                    }
277                })
278                .collect();
279
280            let outputs_count = outputs.len();
281            let outputs_array = quote! {
282                [ #(#outputs),* ]
283            };
284
285            quote! {
286                pub fn #function_name_ident() -> &'static everscale_types::abi::Function {
287                    static ONCE: std::sync::OnceLock<everscale_types::abi::Function> = std::sync::OnceLock::new();
288                    ONCE.get_or_init(|| {
289                        let inputs: [NamedAbiType; #inputs_count] = #inputs_array;
290                        let outputs: [NamedAbiType; #outputs_count] = #outputs_array;
291                        everscale_types::abi::Function::builder(ABI_VERSION, #name)
292                            .with_headers(HEADERS)
293                            .with_inputs(inputs)
294                            .with_outputs(outputs)
295                            .build()
296                    })
297                }
298            }
299        }
300    }

302    fn generate_model(
303        &mut self,
304        name: &str,
305        values: Arc<[NamedAbiType]>,
306    ) -> proc_macro2::TokenStream {
307        let struct_name_ident = format_ident!("{}", name);
308        let mut properties = Vec::<proc_macro2::TokenStream>::new();
309
310        let mut inner_fields = Vec::new();
311
312        let function_tuple = AbiType::Tuple(values.clone());
313
314        for i in values.iter() {
315            let struct_property = match self.unique_tokes.get(&i.ty) {
316                Some(struct_property) => struct_property.clone(),
317                None => self.make_struct_property_with_internal(i.name.to_string(), &i.ty),
318            };
319
320            self.unique_tokes
321                .insert(i.ty.clone(), struct_property.clone());
322
323            inner_fields.push(struct_property.clone());
324
325            let rust_name = i.name.as_ref().to_snake();
326            let rust_property_name_ident = format_ident!("{}", &rust_name);
327
328            let ty_ident = struct_property.type_name_quote();
329
330            let quote = quote! {
331                pub #rust_property_name_ident: #ty_ident,
332            };
333
334            properties.push(quote);
335        }
336
337        if !inner_fields.is_empty() {
338            self.unique_tokes.insert(
339                function_tuple.clone(),
340                StructProperty::Tuple {
341                    name: name.to_string(),
342                    //fields: inner_fields,
343                },
344            );
345        }
346
347        if properties.is_empty() {
348            quote! {
349                type #struct_name_ident = ();
350            }
351        } else {
352            quote! {
353                #[derive(Clone, Debug)]
354                pub struct #struct_name_ident {
355                    #(#properties)*
356                }
357            }
358        }
359    }

361    fn make_function_input_struct(
362        &mut self,
363        name: &str,
364        inputs: Arc<[NamedAbiType]>,
365        is_event: bool,
366    ) -> proc_macro2::TokenStream {
367        let struct_name = if is_event {
368            format!("{}EventInput", name.to_camel())
369        } else {
370            format!("{}EventOutput", name.to_camel())
371        };
372        let model = self.generate_model(&struct_name, inputs.clone());
373
374        if !self.generated_structs.contains_key(&struct_name) {
375            self.generated_structs
376                .insert(struct_name.clone(), inputs.to_vec());
377        }
378
379        model
380    }

389
390        if !self.generated_structs.contains_key(&struct_name) {
391            self.generated_structs
392                .insert(struct_name.clone(), outputs.to_vec());
393        }
394
395        model

406    fn make_struct_property(
407        &mut self,
408        initial_name: Option<String>,
409        param: &AbiType,
410    ) -> StructProperty {
411        let name = initial_name.map(|x| x.to_string());
412        if let Some(st_property) = self.unique_tokes.get(param) {
413            if name
414                .clone()
415                .map(|name| st_property.name().eq(&name))
416                .unwrap_or(false)
417            {
418                return st_property.clone();
419            }
420        }
421
422        match param {
423            AbiType::Uint(a) => {
424                let ty = match a {
425                    8 => "u8",
426                    16 => "u16",
427                    32 => "u32",
428                    64 => "u64",
429                    128 => "u128",
430                    160 => "[u8; 20]",
431                    256 => "everscale_types::prelude::HashBytes",
432                    _ => "num_bigint::BigUint",
433                };
434                StructProperty::Simple {
435                    name,
436                    type_name: syn::parse_str(ty).unwrap(),
437                }
438            }
439            AbiType::Int(a) => {
440                let ty = match a {
441                    8 => "i8",
442                    16 => "i16",
443                    32 => "i32",
444                    64 => "i64",
445                    128 => "i128",
446                    _ => "num_bigint::BigInt",
447                };
448                StructProperty::Simple {
449                    name,
450                    type_name: syn::parse_str(ty).unwrap(),
451                }
452            }
453            AbiType::VarUint(_) | AbiType::VarInt(_) => StructProperty::Simple {
454                name,
455                type_name: syn::parse_quote!(num_bigint::BigUint),
456            },
457            AbiType::Bool => StructProperty::Simple {
458                name,
459                type_name: syn::parse_quote!(bool),
460            },
461            AbiType::Tuple(a) => {
462                let name = name.unwrap_or_default();
463                let camel_case_struct_name = name.to_camel();
464                let struct_name_ident = format_ident!("{}", &camel_case_struct_name);
465
466                let mut structs: Vec<StructProperty> = Vec::new();
467
468                for i in a.iter() {
469                    let property = self.make_struct_property(Some(i.name.to_string()), &i.ty);
470                    structs.push(property);
471                }
472
473                let mut internal_properties: Vec<proc_macro2::TokenStream> = Vec::new();
474
475                for p in &structs {
476                    let p_name = p.name();
477                    let rust_property_name_ident = format_ident!("{}", p_name.as_str().to_snake());
478                    let internal_ident = p.type_name_quote();
479                    let quote = quote! {
480                        pub #rust_property_name_ident: #internal_ident,
481                    };
482                    internal_properties.push(quote);
483                }
484
485                let internal_struct = if !internal_properties.is_empty() {
486                    quote! {
487                        #[derive(Clone, Debug)]
488                        pub struct #struct_name_ident {
489                            #(#internal_properties)*
490                        }
491                    }
492                } else {
493                    quote! {
494                        type #struct_name_ident = ();
495                    }
496                };
497
498                self.temporary_internal_structs_idents.push(internal_struct);
499
500                let property = StructProperty::Tuple {
501                    name,
502                    //fields: structs,
503                };
504
505                {
506                    self.unique_tokes.insert(param.clone(), property.clone());
507                    self.generated_structs
508                        .entry(camel_case_struct_name)
509                        .or_insert_with(|| a.to_vec());
510                }
511                property
512            }
513            AbiType::Array(a) | AbiType::FixedArray(a, _) => {
514                let internal_struct = self.make_struct_property(None, a);
515                StructProperty::Array {
516                    name: name.unwrap_or_default(),
517                    internal: Box::new(internal_struct),
518                }
519            }
520            AbiType::Cell => StructProperty::Simple {
521                name,
522                type_name: syn::parse_quote!(everscale_types::prelude::Cell),
523            },
524            AbiType::Map(a, b) => {
525                let key = match a {
526                    PlainAbiType::Uint(_) | PlainAbiType::Int(_) | PlainAbiType::Address => {
527                        self.make_struct_property(None, &(*a).into())
528                    }
529                    _ => panic!("Map key is not allowed type"),
530                };
531
532                let value = self.make_struct_property(None, b.as_ref());
533
534                StructProperty::HashMap {
535                    name: name.unwrap_or_default(),
536                    key: Box::new(key),
537                    value: Box::new(value),
538                }
539            }
540            AbiType::Address => StructProperty::Simple {
541                name,
542                type_name: syn::parse_quote!(everscale_types::models::message::StdAddr),
543            },
544            AbiType::Bytes | AbiType::FixedBytes(_) => StructProperty::Simple {
545                name,
546                type_name: syn::parse_quote!(Vec<u8>),
547            },
548            AbiType::String => StructProperty::Simple {
549                name,
550                type_name: syn::parse_quote!(String),
551            },
552            AbiType::Token => StructProperty::Simple {
553                name,
554                type_name: syn::parse_quote!(everscale_types::num::Tokens),
555            },
556            AbiType::Optional(a) => {
557                let internal_struct = self.make_struct_property(None, a.as_ref());
558                StructProperty::Option {
559                    name: name.unwrap_or_default(),
560                    internal: Box::new(internal_struct),
561                }
562            }
563            AbiType::Ref(a) => self.make_struct_property(name, a.as_ref()),
564        }
565    }
566}

568#[derive(Clone)]
569enum StructProperty {
570    Simple {
571        name: Option<String>,
572        type_name: syn::Type,
573    },
574    Tuple {
575        name: String,
576        //_fields: Vec<StructProperty>,
577    },
578    Array {
579        name: String,
580        internal: Box<StructProperty>,
581    },
582    Option {
583        name: String,
584        internal: Box<StructProperty>,
585    },
586    HashMap {
587        name: String,
588        key: Box<StructProperty>,
589        value: Box<StructProperty>,
590    },
591}
592
593impl StructProperty {
594    pub fn type_name_quote(&self) -> syn::Type {
595        match self {
596            StructProperty::Simple { type_name, .. } => type_name.clone(),
597            StructProperty::Tuple { name, .. } => syn::parse_str(&name.to_camel()).unwrap(),
598            StructProperty::Array { internal, .. } => {
599                let ty = internal.type_name_quote();
600                syn::parse_quote!(Vec<#ty>)
601            }
602            StructProperty::Option { internal, .. } => {
603                let ty = internal.type_name_quote();
604                syn::parse_quote!(Option<#ty>)
605            }
606            StructProperty::HashMap { key, value, .. } => {
607                let key = key.type_name_quote();
608                let value = value.type_name_quote();
609                syn::parse_quote!(std::collections::HashMap<#key, #value>)
610            }
611        }
612    }
613
614    pub fn name(&self) -> String {
615        match self {
616            StructProperty::Simple { name, .. } => {
617                let name = name.clone();
618                name.unwrap_or_default()
619            }
620            StructProperty::Tuple { name, .. } => name.clone(),
621            StructProperty::Array { name, .. } => name.clone(),
622            StructProperty::Option { name, .. } => name.clone(),
623            StructProperty::HashMap { name, .. } => name.clone(),
624        }
625    }
626}

628fn quote_abi_type(ty: &AbiType) -> proc_macro2::TokenStream {
629    let quote: proc_macro2::TokenStream = match ty.clone() {
630        AbiType::String => {
631            let ty: syn::Type = syn::parse_quote!(everscale_types::abi::AbiType::String);
632            quote! {
633                #ty
634            }
635        }
636        AbiType::Address => {
637            let ty: syn::Type = syn::parse_quote!(everscale_types::abi::AbiType::Address);
638            quote! {
639                #ty
640            }
641        }
642        AbiType::Bool => syn::parse_quote!(everscale_types::abi::AbiType::Bool),
643        AbiType::Bytes => syn::parse_quote!(everscale_types::abi::AbiType::Bytes),
644        AbiType::FixedBytes(size) => {
645            syn::parse_quote!(everscale_types::abi::AbiType::FixedBytes(#size))
646        }
647        AbiType::Cell => syn::parse_quote!(everscale_types::abi::AbiType::Cell),
648        AbiType::Token => syn::parse_quote!(everscale_types::abi::AbiType::Token),
649        AbiType::Int(value) => quote! {
650            everscale_types::abi::AbiType::Int(#value)
651        },
652        AbiType::Uint(value) => {
653            quote! {
654                everscale_types::abi::AbiType::Uint(#value)
655            }
656        }
657        AbiType::VarInt(value) => {
658            let val = value.get();
659            quote! {
660                everscale_types::abi::AbiType::Int(core::num::nonzero::NonZeroU8(#val))
661            }
662        }
663        AbiType::VarUint(value) => {
664            let val = value.get();
665            quote! {
666                everscale_types::abi:AbiType::Uint(core::num::nonzero::NonZeroU8(#val))
667            }
668        }
669        AbiType::Tuple(tuple) => {
670            let mut tuple_properties = Vec::new();
671
672            for i in tuple.iter() {
673                let name_abi_quote = make_abi_type(i.name.as_ref(), i.ty.clone());
674                tuple_properties.push(name_abi_quote);
675            }
676
677            quote! {
678                everscale_types::abi::AbiType::Tuple(std::sync::Arc::new([ #(#tuple_properties),*]))
679            }
680        }
681        AbiType::Array(ty) => {
682            let ty = quote_abi_type(&ty);
683            quote! {
684                everscale_types::abi::AbiType::Array(std::sync::Arc::new(#ty))
685            }
686        }
687        AbiType::FixedArray(ty, size) => {
688            let ty = quote_abi_type(&ty);
689            quote! {
690                everscale_types::abi:AbiType::FixedArray(std::sync::Arc<#ty>, #size)
691            }
692        }
693        AbiType::Map(key, value) => {
694            let key_type: proc_macro2::TokenStream = match key {
695                PlainAbiType::Address => {
696                    let ty: syn::Type =
697                        syn::parse_quote!(everscale_types::abi::PlainAbiType::Address);
698                    quote! {
699                        #ty
700                    }
701                }
702                PlainAbiType::Bool => {
703                    let ty: syn::Type = syn::parse_quote!(everscale_types::abi::PlainAbiType::Bool);
704                    quote! {
705                        #ty
706                    }
707                }
708                PlainAbiType::Uint(value) => {
709                    quote! {
710                        everscale_types::abi::PlainAbiType::Uint(#value)
711                    }
712                }
713                PlainAbiType::Int(value) => {
714                    quote! {
715                        everscale_types::abi::PlainAbiType::Int(#value)
716                    }
717                }
718            };
719
720            let value_type = quote_abi_type(&value);
721            syn::parse_quote!(everscale_types::abi::AbiType::Map(#key_type, std::sync::Arc::new(#value_type)))
722        }
723        AbiType::Optional(_) => {
724            let ty = quote_abi_type(ty);
725            quote! {
726                everscale_types::abi::AbiType::Optional(std::sync::Arc<#ty>)
727            }
728        }
729        AbiType::Ref(_) => {
730            let ty = quote_abi_type(ty);
731            quote! {
732                everscale_types::abi::AbiType::Ref(std::sync::Arc<#ty>)
733            }
734        }
735    };
736    quote
737}
738
739fn quote_abi_value(name: &str) -> proc_macro2::TokenStream {
740    let name_ident = format_ident!("{}", name.to_snake());
741    quote! {
742        everscale_types::abi::IntoAbi::as_abi(&self.#name_ident)
743    }
744}
745
746fn make_abi_type(name: &str, abi_type: AbiType) -> proc_macro2::TokenStream {
747    let abi_type = quote_abi_type(&abi_type);
748
749    quote! {
750        NamedAbiType::new(#name, #abi_type)
751    }
